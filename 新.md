## 最近公共祖先（LCA）

**节点编号0/1开始都是可以的**

需要维护1.`fa[][]`倍增数组     2.`dep[ ]`深度数组

```c++
const int N = 5e5 + 10;
int n, m, s, a, b;
vector<int> e[N];  // 邻接表存储图，e[i]存储与节点i相连的节点
int dep[N];       // dep[i]表示节点i的深度
int fa[N][20];    // fa[i][j]表示节点i的第2^j个祖先
//int dis[N];       //每个点到根节点的距离

// 深度优先搜索函数，用于初始化节点的深度和祖先信息
void dfs(int u, int father, int val) {
    dep[u] = dep[father] + 1;  // 设置节点u的深度为其父节点father深度加1
    fa[u][0] = father;         // 节点u的第2^0（即1）个祖先为其父节点father
    //dis[u]=dis[father]+val;  //计算每个点到根节点的距离
    
    for (int i = 1; i <= 19; i++) { 
        // TOP为2^j中j的范围，根据题目范围决定，当然也可以无脑开的稍微大一些
        fa[u][i] = fa[fa[u][i - 1]][i - 1]; 
        // 节点u的第2^i个祖先，等于节点u的第2^(i-1)个祖先的第2^(i-1)个祖先
    }
    for (int v : e[u]) {  	// 遍历与节点u相连的节点v
        if (v != father) {  // 如果v不是u的父节点
            dfs(v, u);     	// 递归处理节点v，u作为v的父节点
        }
    }
}

// 计算最近公共祖先（LCA）的函数					
int lca(int u, int v) {
    //一、u,v到达同一深度
    if (dep[u] < dep[v]) {  // 如果u的深度小于v的深度，交换u和v，保证u的深度不小于v的深度
        swap(u, v);
    }
    for (int i = 19; i >= 0; i--) {
        if (dep[fa[u][i]] >= dep[v]) {  // 从大到小枚举u的祖先，找到深度{不大于}v的祖先
            u = fa[u][i];  // 将u更新为该祖先      //最终跳到同一层
        }
    }
    if (u == v) {  // 如果u和v相等，说明v就是它们的最近公共祖先
        return v;
    }
    // 二、找到共同祖先
    for (int i = 19; i >= 0; i--) {  // 从大到小枚举u和v的祖先
        if (fa[u][i] != fa[v][i]) {  // 如果u和v的第2^i个祖先{不相等}
            u = fa[u][i];  // u更新为第2^i个祖先
            v = fa[v][i];  // v更新为第2^i个祖先
        }
    }
    return fa[u][0];  // 返回u的父节点，即为u和v的最近公共祖先
}
ll Dis(ll a,ll b){
	return dis[a]+dis[b]-2*dis[lca(a,b)];
}
```

## 线性基

线性基是**用于处理异或问题**的一种数据结构，核心就一句话：

用**最少的一组数**，能异或出原集合能异或出的**所有结果**。

1. **只针对异或**：不是加法、乘法，是**按位异或**

2. 性质

   - 线性基里的数，**互相异或不会得到 0**
   - 原集合任意数，都能被线性基里的数**异或出来**
   - 线性基**大小最小**，没有冗余
   - 异或和个数 2^n^ -1

3. 作用

   - 求一组数的**最大异或和**
   - 求能异或出多少种不同值
   - **判断某个数能不能被异或出来**

   ```c++
   const int MAX_BIT = 63;  // 处理long long范围(0~63位)，int的话用31即可
   struct LinearBasis {
       ll d[MAX_BIT + 1];  // d[i]表示第i位为1的线性基元素
       bool flag;          // 标记原集合是否能异或出0（存在重复/线性相关）
   
       // 初始化线性基
       LinearBasis() {
           fill(d, d + MAX_BIT + 1, 0);
           flag = false;
       }
   
       // 插入一个数到线性基
       void insert(ll x) {
           // 从高位到低位遍历
           for (int i = MAX_BIT; i >= 0; --i) {
               if (x & (1LL << i)) {  // 找到x的最高非零位
                   if (!d[i]) {       // 该位线性基为空，直接插入
                       d[i] = x;
                       break;
                   } else {           // 该位已有基，异或消去该位
                       x ^= d[i];
                   }
               }
           }
           // 如果x最终变为0，说明原集合存在线性相关，能异或出0
           if (x == 0) flag = true;
       }
   
       // 求最大异或和
       ll get_max() {
           ll res = 0;
           // 从高位到低位贪心，尽可能让每一位为1
           for (int i = MAX_BIT; i >= 0; --i) {
               if ((res ^ d[i]) > res) {
                   res ^= d[i];
               }
           }
           return res;
       }
   
       // 求最小异或和
       ll get_min() {
           // 如果能异或出0，最小就是0
           if (flag) return 0;
           // 否则找线性基中最小的元素（最低位的非零基）
           for (int i = 0; i <= MAX_BIT; ++i) {
               if (d[i]) return d[i];
           }
           return 0;  // 空集情况
       }
   
       // 判断数x是否能被线性基异或出来
       bool check(ll x) {
           if (x == 0) return flag;  // 0需要看是否能异或出
           for (int i = MAX_BIT; i >= 0; --i) {
               if (x & (1LL << i)) {
                   x ^= d[i];
                   if (x == 0) return true;
               }
           }
           return false;
       }
   };
   
   // 测试样例
   int main() {
       LinearBasis lb;
       // 插入一组数
       ll a[] = {1, 2, 4, 8};
       int n = sizeof(a) / sizeof(ll);
       for (int i = 0; i < n; ++i) {
           lb.insert(a[i]);
       }
   
       cout << "最大异或和: " << lb.get_max() << endl;  // 输出15 (1^2^4^8)
       cout << "最小异或和: " << lb.get_min() << endl;  // 输出1
       cout << "能否异或出3: " << (lb.check(3) ? "是" : "否") << endl;  // 是（1^2）
       cout << "能否异或出5: " << (lb.check(5) ? "是" : "否") << endl;  // 是（1^4）
       cout << "能否异或出7: " << (lb.check(7) ? "是" : "否") << endl;  // 是（1^2^4）
       cout << "能否异或出0: " << (lb.check(0) ? "是" : "否") << endl;  // 否
   
       return 0;
   }
   ```

## 树状数组上二分

 **整体二分的应用场**

1. 单个问题的答案具有单调性，当答案固定时，问题的要求可能得到满足或者无法满足
2. 如果要求得到满足，那么答案变差，要求依然能满足，此时去寻找更好的答案
3. 如果要求无法满足，那么答案变好，要求也无法满足，此时去寻找更差的答案
4. 问题数量较多，无法对每个问题都执行一次二分答案，但是所有查询允许**离线处理**

**整体二分的基本思想**

1. 让修改操作适应当前的答案范围，调整好数据状况，然后查看每个问题的要求是否达标
2. **多个问题共享这一轮的修改操作后的数据状况，避免重复劳动**
3. 达标的问题去往左组，不达标的问题去往右组，然后左右分别递归，直到解决所有的问题

**整体二分的时间复杂度**

​	假设，答案范围为`n`，问题数量为`q`，单次修改或查询为`log n`

​	那么，总的时间复杂度为O((n + q) * log n * log n)

​	O((n + q) * log n ***数据结构的单次修改或查询复杂度**)

```c++
    build(ls[u],l,mid);
    build(rs[u],mid+1,r);
```

```c++
    u=++tot; //动态开点
    ls[u]=ls[v]; rs[u]=rs[v]; sum[u]=sum[v]+1;//构造新节点
    if(l==r) return;  //双指针同步搜索

    if(p<=mid) change(ls[u],ls[v],l,mid,p);

    if(l==r)return l;  //双指针同步搜索
    int s=sum[ls[u]]-sum[ls[v]];
    int mid=(l+r)>>1;

int query_range(int u, int v, int l, int r, int L, int R) {
    if (!v|| L > R) return 0;
    if (L <= l && r <= R) return ls[v] - rs[u];
    int mid = (l + r) >> 1, res = 0;
    if (L <= mid) res += query_range(ls[u], ls[v], l, mid, L, R);
    if (R > mid) res += query_range(rs[u], rs[v], mid + 1, r, L, R);
    return res;
}
```

```c++
int root[N],idx;//根节点
void build(int &x,int l,int r){//空树
    x=++idx;//编号
    if(l==r)return;
    int m=l+r>>1;
    build(lc(x),l,m);
    build(rc(x),m+1,m);
}
void insert(int x,int &y,int l,int r,int v){//x前版本根，y新版本根
    y=++idx;tr[y]=tr[x];tr[y].s++;//编号，新节点赋值原节点
    if(l==r)return;
    int m=l+r>>1;
    if(v<=m)insert(lc(x),lc(y),l,m,v);
    else insert(rc(x),rc(y),m+1,r,v);
}
int query(int x,int y,int l,int r,int k){////查询[l~r]第k小的数
    if(l==r)return 1;
    int m=l+r>>1;
    int s=tr[lc(y)].s-tr[lc(x)].s;//前缀和，相同范围区间
    if(k<=s)return query(lc(x),lc(y),l,m,k);
    else return query(rc(x),rc(y),m+1,r,k-s);
}
// 查询区间[l,r]中值在[L, R]范围内的点数
int query_range(int x, int y, int l, int r, int L, int R) {
    if (!y || L > R) return 0;
    if (L <= l && r <= R) return tr[y].s - tr[x].s;
    int mid = (l + r) >> 1, res = 0;
    if (L <= mid) res += query_range(lc(x), lc(y), l, mid, L, R);
    if (R > mid) res += query_range(rc(x), rc(y), mid + 1, r, L, R);
    return res;
}
void solve(){
    build(root[0],1,n);
    for(int i=1;i<=n;i++)insert(root[i-1],root[i],1,n,a[i]);
    cin>>l>>r;
    query(root[l-1],root[r],l,r,k);//查询[l~r]第k小的数
}
```

## 线段树上二分

美味度排序，**整体二分美味度**，用线段树维护（单价线段树 查询最小花费）是否满足条件

```c++
const int N=1e5+5,mod=998244353,INF=1e18;
int n,m;
struct node{
    int d,p,l;
};
node juice[N];
int gid[N],money[N],least[N],ans[N];
int used=0,maxp=0;
int cost[N<<2];
int suml[N<<2];
int lset[N],rset[N];
void up(int root){
    suml[root]=suml[root<<1]+suml[root<<1|1];
    cost[root]=cost[root<<1]+cost[root<<1|1];
}
void add(int root, int l, int r, int p, int val){
    if(l == r){
        suml[root] += val;
        cost[root] = l * suml[root];
        return;
    }
    int mid = (l + r) / 2;
    if(p <= mid) add(root<<1, l, mid, p, val);
    else add(root<<1|1, mid+1, r, p, val);
    up(root);
}

long long query(int root, int l, int r, long long need) {
    if (l == r) {
        return need * l;
    }
    int mid = (l + r) / 2;
    if (suml[root << 1] >= need) {           
        return query(root << 1, l, mid, need);
    } else {
        return cost[root << 1] + query(root << 1 | 1, mid + 1, r, need - suml[root << 1]);
    }
}
void compute(int ql,int qr,int vl,int vr){
    if(ql>qr)return;
    if(vl==vr){
        for(int i=ql;i<=qr;i++){
            ans[gid[i]]=vl;
        }
    }else{
        int mid=(vl+vr)/2;
        while(used<mid){
            used++;
            add(1,1,maxp,juice[used].p,juice[used].l);
        }
        while(used>mid){
            add(1,1,maxp,juice[used].p,-juice[used].l);
            used--;
        }
        int lsiz=0,rsiz=0;
        for(int i=ql;i<=qr;i++){
            int id=gid[i];
            if(suml[1]>=least[id]&&query(1,1,maxp,least[id])<=money[id]){
                lset[++lsiz]=id;
            }else{
                rset[++rsiz]=id;
            }
        }
        for(int i=1;i<=lsiz;i++){
            gid[ql+i-1]=lset[i];
        }
        //  唯一修改：这里把 rsiz 改成 lsiz！
        for(int i=1;i<=rsiz;i++){
            gid[ql + lsiz + i - 1] = rset[i];
        }
        compute(ql,ql+lsiz-1,vl,mid);
        compute(ql+lsiz,qr,mid+1,vr);
    }
}
void solve(){
    cin>>n>>m;
    for(int i=1;i<=n;i++){
        int d,p,l;cin>>d>>p>>l;
        juice[i]={d,p,l};
        maxp=max(maxp,p);
    }
    sort(juice+1,juice+n+1,[&](node a,node b){
        return a.d>b.d;
    });
    for(int i=1;i<=m;i++){
        gid[i]=i;
        cin>>money[i]>>least[i];
    }
    compute(1,m,1,n+1);
    for(int i=1;i<=m;i++){
        if(ans[i]==n+1){
            cout<<-1<<'\n';
        }else{
            cout<<juice[ans[i]].d<<'\n';
        }
    }
}
```
