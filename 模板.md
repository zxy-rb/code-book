在不同的编程语言和场景中，`cmp`（比较函数）返回值的含义以及是否会基于此进行交换操作有所不同。下面以常见的情况为例进行说明。

在 Python 2 里，像 `sorted`、`list.sort` 这类排序函数支持传入一个 `cmp` 参数，它是一个比较函数。该函数接收两个参数，若第一个参数小于第二个参数，就返回负数；若相等，返回 0；若第一个参数大于第二个参数，返回正数。排序时，若 `cmp` 函数返回正数，就会交换这两个元素的位置。

示例代码如下：
```python
def custom_cmp(x, y):
    # 自定义比较规则，这里简单认为 x 大于 y 时返回正数
    if x > y:
        return 1
    elif x < y:
        return -1
    return 0

numbers = [3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5]
numbers.sort(cmp=custom_cmp)
print(numbers)

```
不过 Python 3 摒弃了 `cmp` 参数，而是采用 `key` 参数。

在 C++ 中，`std::sort` 这类排序函数可以传入一个比较函数或者函数对象。比较函数要返回布尔值，若返回 `true`，就表示第一个参数应该排在第二个参数前面，不会进行交换；若返回 `false`，就会交换这两个元素的位置。

示例代码如下：
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

// 自定义比较函数
bool custom_cmp(int x, int y) {
    // 自定义比较规则，这里简单认为 x 小于 y 时返回 true
    return x < y;
}

int main() {
    std::vector<int> numbers = {3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5};
    std::sort(numbers.begin(), numbers.end(), custom_cmp);
    for (int num : numbers) {
        std::cout << num << " ";
    }
    std::cout << std::endl;
    return 0;
}
```

```c++
int partition(vector<int>& a,int low ,int high){
    int pivot =a[low];
    while(low<high){
        while(low<high&&a[high]>=pivot)high--;
        a[low]=a[high];
        while(low<high&&a[low]<=pivot)low++;
        a[high]=a[low];
    }
    a[high]=pivot;
    return high;
}
void quicksort(vector<int>& a,int low ,int high){
    if(low<high){
        int pi=partition(a,low,high);

        quicksort(a,pi+1,high);

}
```

```c++
void merge(vector<int>&a,int l,int mid,int r){
    int n1=mid-l+1;
    int n2=r-mid;
    vector<int>L(n1),R(n2);
    for(int i=0;i<n1;i++)
        L[i]=a[l+i];
    for(int i=0;i<n2;i++)
        R[i]=a[mid+1+i];
    int i=0,j=0,k=l;
    while(i<n1 && j<n2){
        if(L[i]<R[j]){
            a[k]=L[i];
            i++;
        }else{
            a[k]=R[j];
            j++;
        }
        k++;
    }
    while(i<n1){
        a[k]=L[i];
        i++;k++;
    }
    while(j<n2){
        a[k]=R[j];
        j++;k++;
    }
}
void merge_sort(vector<int>&a, int l, int r) {
    if (l < r ) {
        // 分解
        int mid = l + ((r - l) >> 1);
        merge_sort(a, l, mid);
        merge_sort(a, mid+1, r);
        // 合并
        merge(a, l, mid, r); 
    }
}
```

​	差分：**很少的询问**，但有**很多改变数组**的操作，每次让不同区间[l，r]内的每个数都加c或者减c

​	适用场景：长度为n的数组a，1次询问，m次操作，每次令区间[l，r]的每个数+c或-c。

```c++
const int N=1e5+10;

    int n,m; cin>>n>>m;
    for(int i=1;i<=n;i++) {
        cin>>a[i];
        d[i]=a[i]-a[i-1];  //求差分数组
    }
    while(m--){
        int l,r,c;
        cin>>l>>r>>c;
        d[l]+=c;  //差分标记
        d[r+1]-=c;
    }
    for(int i=1;i<=n;i++){
```

### 期望

在概率论和统计学中，“期望”（也叫均值、数学期望）是描述随机变量平均取值的指标，反映了随机变量取值的 “中心趋势”。计算方法根据随机变量的类型（离散型、连续型）有所不同，核心思想是 “**取值乘以对应概率的加权平均**”。

- 离散型：逐个取值乘以对应概率，再求和；

## 线性基

线性基是**用于处理异或问题**的一种数据结构，核心就一句话：

用**最少的一组数**，能异或出原集合能异或出的**所有结果**。

1. **只针对异或**：不是加法、乘法，是**按位异或**

2. 性质

   - 线性基里的数，**互相异或不会得到 0**
   - 原集合任意数，都能被线性基里的数**异或出来**
   - 线性基**大小最小**，没有冗余

3. 作用

   - 求一组数的**最大异或和**
   - 求能异或出多少种不同值
   - **判断某个数能不能被异或出来**

   ```c++
   const int MAX_BIT = 63;  // 处理long long范围(0~63位)，int的话用31即可
   struct LinearBasis {
       ll d[MAX_BIT + 1];  // d[i]表示第i位为1的线性基元素
       bool flag;          // 标记原集合是否能异或出0（存在重复/线性相关）
   
       // 初始化线性基
       LinearBasis() {
           fill(d, d + MAX_BIT + 1, 0);
           flag = false;
       }
   
       // 插入一个数到线性基
       void insert(ll x) {
           // 从高位到低位遍历
           for (int i = MAX_BIT; i >= 0; --i) {
               if (x & (1LL << i)) {  // 找到x的最高非零位
                   if (!d[i]) {       // 该位线性基为空，直接插入
                       d[i] = x;
                       break;
                   } else {           // 该位已有基，异或消去该位
                       x ^= d[i];
                   }
               }
           }
           // 如果x最终变为0，说明原集合存在线性相关，能异或出0
           if (x == 0) flag = true;
       }
   
       // 求最大异或和
       ll get_max() {
           ll res = 0;
           // 从高位到低位贪心，尽可能让每一位为1
           for (int i = MAX_BIT; i >= 0; --i) {
               if ((res ^ d[i]) > res) {
                   res ^= d[i];
               }
           }
           return res;
       }
   
       // 求最小异或和
       ll get_min() {
           // 如果能异或出0，最小就是0
           if (flag) return 0;
           // 否则找线性基中最小的元素（最低位的非零基）
           for (int i = 0; i <= MAX_BIT; ++i) {
               if (d[i]) return d[i];
           }
           return 0;  // 空集情况
       }
   
       // 判断数x是否能被线性基异或出来
       bool check(ll x) {
           if (x == 0) return flag;  // 0需要看是否能异或出
           for (int i = MAX_BIT; i >= 0; --i) {
               if (x & (1LL << i)) {
                   x ^= d[i];
                   if (x == 0) return true;
               }
           }
           return false;
       }
   };
   
   // 测试样例
   int main() {
       LinearBasis lb;
       // 插入一组数
       ll a[] = {1, 2, 4, 8};
       int n = sizeof(a) / sizeof(ll);
       for (int i = 0; i < n; ++i) {
           lb.insert(a[i]);
       }
   
       cout << "最大异或和: " << lb.get_max() << endl;  // 输出15 (1^2^4^8)
       cout << "最小异或和: " << lb.get_min() << endl;  // 输出1
       cout << "能否异或出3: " << (lb.check(3) ? "是" : "否") << endl;  // 是（1^2）
       cout << "能否异或出5: " << (lb.check(5) ? "是" : "否") << endl;  // 是（1^4）
       cout << "能否异或出7: " << (lb.check(7) ? "是" : "否") << endl;  // 是（1^2^4）
       cout << "能否异或出0: " << (lb.check(0) ? "是" : "否") << endl;  // 否
   
       return 0;
   }
   ```

### 裴蜀定理

设 a,b 为**不全为零**的整数，记 d=gcd(a,b)（a 与 b 的最大公约数），则：

​	1.**存在性**：一定存在整数 x,y，使得 ax+by=d

​	2.**最小性**：d 是 a 与 b 的所有整系数线性组合 ax+by 中，能表示出的**最小正整数**。

重要推论（方程可解性）对于线性不定方程 ax+by=c（a,b,c 为整数，a,b 不全为零）：

**方程有整数解 x,y) 的充要条件是**：

​	**gcd(a,b)∣c**    即 c 必须是 a,b 最大公约数 d 的倍数。

**特例**：方程 ax+by=1 有整数解 ⟺gcd(a,b)=1（即 a 与 b **互质**）。

求裴蜀定理中的x，y的方法又叫做**扩展欧几里得算法**。

```c++
int x,y;
ll exgcd(ll a, ll b, ll &x, ll &y) {
    if (b == 0) {
        x = 1;
        y = 0;
        return a;
    }
    // 递归计算gcd(b, a%b)，得到下一层的解x', y'
    ll d = exgcd(b, a % b, x, y);
    // 保存当前x'（即下一层的x），用于推导当前层的y
    ll next= x;
    // 推导当前层的x和y：x = y', y = x' - (a/b)*y'
    x = y;
    y = next - (a / b) * y;
    // 返回最大公约数
    return d;
}
```

在计算组合数 ***C*(*n*,*k*)=*n*!  / *k*!(*n*−*k*)!** 时，若题目要求结果对某个质数 *m* 取模，直接做除法可能会出错，这时就需要借助**逆元**来解决问题。下面为你详细介绍逆元在计算组合数时的运用。

### 计算组合数取模

**核心思路**

取模的公式为：
$$
C(n, k) \mod m = \frac{n! \mod m}{(k! \cdot (n-k)!) \mod m} \mod m
$$
 但在模运算里，不能直接进行除法操作，需要把除法转换为乘以逆元，即： \(
$$
\frac{a}{b} \mod m = a \cdot b^{-1} \mod m
$$
\) 这里的 `(b^-1)` 是 b 在模 m 下的乘法逆元。

**计算步骤**

1. **预处理阶乘模 m**：提前计算出 \(0!\) 到 \(n!\) 每个数对 m 取模后的结果。

2. **计算逆元**：算出 `k! (n-k)!` 在模 m 下的**逆元**。
   
3. **计算结果**：利用公式 \(
   $$
   C(n, k) \mod m = n! \cdot (k! \cdot (n-k)!)^{-1} \mod m
   $$
   ​						`(n!*inv[k])%mod*inv[n-k]%mod;`        (inv[x]表示x关于m的逆元) 

   ) 得出最终结果。

   ### 方法一：费马小定理计算单个组合数

   当 m 是质数时，根据费马小定理，b 的逆元为 **\(b^{m-2} mod m\)**。

```cpp

    ll res = 1;

        a = a * a % mod;

ll C(ll n, ll k) {
    if (k < 0 || k > n) return 0;
    
    // 计算分子 n!
    ll x = 1;
    for (ll i = 1; i <= n; i++) {
        x = x * i % mod;

    ll y = 1;

        y = y * i % mod;
    }
    for (ll i = 1; i <= n - k; i++) {
        y = y * i % mod;
    }
    
    // 使用费马小定理计算逆元
    ll inv = pow_m(y, mod - 2);
    return x * inv % mod;
```

## 逆元

**定义**
    逆元又称为数论倒数。这里和普通的倒数是不一样的，普通倒数可能为小数，而数论倒数一定是个整数。其定义为：对于正整数 `aa` 和 `pp`，如果有 `ax≡1(modp)`，那么把这个同余方程中x的最小正整数解叫做a模p的逆元；即ax % p = 1中x的最小正整数解就是a关于模数p的逆元。

​									**inv（a）= a ^（mod - 2）**

**背景**
    由模运算可知，部分取模运算满足运算律，其形式如下：

```c++
(a  +  b) % p = (a%p + b%p) %p  	（对）
(a  -  b) % p = (a%p - b%p) %p  	（对）
(a  *  b) % p = (a%p * b%p) %p  	（对）
(a  /  b) % p = (a%p / b%p) %p  	（错）
```

​        取模运算对于除法是不成立的，那么如果模运算中出现了除法是不是就无法计算了呢？伟大的数学家于是给出了逆元这个东西。已知 ax % p = 1，这里x不一定等于 1 / a ， 因为加入了取模；比如 3 * 2%5 = 1 ， 这里的2不就相当于 1 / 3的作用吗？于是我们就说2是3关于模数5的逆元，这就是逆元产生的意义。于是，我们在**计算除法取模**时，可以把**除数变为他的逆元**，把**除法取模运算变为乘法取模运算**。在二式等价背景下，变除为乘，间接实现取模运算，这就是逆元的魅力！（后文均用inv(a)代表a的逆元）

```c++
    由上可知： （a/b）%p  = （a * inv（b））%p = （a%p * inv（b）%p）%p，即除以一个数取模等于乘以这个数的逆元取模。
```

```c++
void dfs(int u, int father, int val) {
    dep[u] = dep[father] + 1;  // 设置节点u的深度为其父节点father深度加1
    fa[u][0] = father;         // 节点u的第2^0（即1）个祖先为其父节点father

        fa[u][i] = fa[fa[u][i - 1]][i - 1];

    }

        if (v != father) {  // 如果v不是u的父节点
            dfs(v, u);     	// 递归处理节点v，u作为v的父节点
        }
    }
}

// 计算最近公共祖先（LCA）的函数					
int lca(int u, int v) {
    //一、u,v到达同一深度
    if (dep[u] < dep[v]) {  // 如果u的深度小于v的深度，交换u和v，保证u的深度不小于v的深度
        swap(u, v);
    }
    for (int i = 19; i >= 0; i--) {
        if (dep[fa[u][i]] >= dep[v]) {  // 从大到小枚举u的祖先，找到深度{不大于}v的祖先
            u = fa[u][i];  // 将u更新为该祖先      //最终跳到同一层
        }
    }
    if (u == v) {  // 如果u和v相等，说明v就是它们的最近公共祖先
        return v;
    }
    // 二、找到共同祖先
    for (int i = 19; i >= 0; i--) {  // 从大到小枚举u和v的祖先
        if (fa[u][i] != fa[v][i]) {  // 如果u和v的第2^i个祖先{不相等}
            u = fa[u][i];  // u更新为第2^i个祖先
            v = fa[v][i];  // v更新为第2^i个祖先
        }
    }
    return fa[u][0];  // 返回u的父节点，即为u和v的最近公共祖先
}
ll Dis(ll a,ll b){
	return dis[a]+dis[b]-2*dis[lca(a,b)];
}
```

## 强连通分量 Tarjan

**核心概念**

1. **时间戳 `dfn[x]`**：节点 `x` 第一次被访问的顺序。
2. **追溯值 `low[x]`**：以 x 为根的子树中，所能到达的最早的**时间戳**。

**算法步骤**

1. **入 x 时**：给 `x` 盖时间戳、将 `x` 入栈。

2. **枚举 x 的邻点 y**，分三种情况处理：

   (1) 若 `y` 尚未访问：对 `y` 深度搜索。**回溯到 x 时**，用 `low[y]` 更新 `low[x]`（因为 x 是 y 的父节点，y 能访问到的点，x 一定也能访问到）。

   (2) 若 `y` 已访问且在栈中：说明 `y` 是祖先节点或左子树节点，用 `dfn[y]` 更新 `low[x]`。

   (3) 若 `y` 已访问且不在栈中：说明 `y` 已搜索完毕，其所在连通分量已被处理，无需操作。

1. 离开 x 时，（**出边**全都处理好了）

   ​       记录强连通分量（SCC）。只有遍历完一个 SCC，才可以出栈。

   - 更新 `low` 值的意义：**避免 SCC 的节点提前出栈**。（low是这组强连通分量的最早根，最大环）

```c++
vector<int> e[N];
```

​	给你一个 正 整数 `n` ，请你返回区间 `[1, n]` 之间（每一个数位都是 互不相同的整数）的数目。

```c++
class Solution {
public:
    int countSpecialNumbers(int n) {
        string s = to_string(n);
        int m = s.length();
        vector<vector<int>> memo(m, vector<int>(1 << 10, -1)); // -1 表示没有计算过
        auto dfs = [&](auto&& dfs, int i, int mask, bool is_limit, bool is_num) -> int {
            if (i == m) {
                return is_num; // is_num 为 true 表示得到了一个合法数字
            }
            if (!is_limit && is_num && memo[i][mask] != -1) {
                return memo[i][mask]; // 之前计算过
            }
            int res = 0;
            if (!is_num) { // 可以跳过当前数位
                res = dfs(dfs, i + 1, mask, false, false);
            }
            // 如果前面填的数字都和 n 的一样，那么这一位至多填数字 s[i]（否则就超过 n 啦）
            int up = is_limit ? s[i] - '0' : 9;
            // 枚举要填入的数字 d
            // 如果前面没有填数字，则必须从 1 开始（因为不能有前导零）
            for (int d = is_num ? 0 : 1; d <= up; d++) {
                if ((mask >> d & 1) == 0) { // d 不在 mask 中，说明之前没有填过 d
                    res += dfs(dfs, i + 1, mask | (1 << d), is_limit && d == up, true);
                }
            }
            if (!is_limit && is_num) {
                memo[i][mask] = res; // 记忆化
            }
            return res;
        };
        return dfs(dfs, 0, 0, true, false);
    }
};
```

```c++
// 代码示例：返回 [low, high] 中的恰好包含 target 个 0 的数字个数
// 比如 digitDP(0, 10, 1) == 2
// 要点：我们统计的是 0 的个数，需要区分【前导零】和【数字中的零】，前导零不能计入，而数字中的零需要计入
long long digitDP(long long low, long long high, int target) {
    string low_s = to_string(low);
    string high_s = to_string(high);
    int n = high_s.size();
    int diff_lh = n - low_s.size();
    vector memo(n, vector<long long>(target + 1, -1));
    auto dfs = [&](this auto&& dfs, int i, int cnt0, bool limit_low, bool limit_high) -> long long {
        if (cnt0 > target) {
            return 0; // 不合法
        }
        if (i == n) {
            return cnt0 == target;
        }
        if (!limit_low && !limit_high && memo[i][cnt0] >= 0) {
            return memo[i][cnt0];
        }
        int lo = limit_low && i >= diff_lh ? low_s[i - diff_lh] - '0' : 0;
        int hi = limit_high ? high_s[i] - '0' : 9;
        long long res = 0;
        int d = lo;
        // 通过 limit_low 和 i 可以判断能否不填数字，无需 is_num 参数
        // 如果前导零不影响答案，去掉这个 if block
        if (limit_low && i < diff_lh) {
            // 不填数字，上界不受约束
            res = dfs(i + 1, 0, true, false);
            d = 1;
        }
        for (; d <= hi; d++) {
            // 统计 0 的个数
            res += dfs(i + 1, cnt0 + (d == 0), limit_low && d == lo, limit_high && d == hi);
            // res %= MOD;
        }
        if (!limit_low && !limit_high) {
            memo[i][cnt0] = res;
        }
        return res;
    };
    return dfs(0, 0, true, true);
}
```

### 树状数组上二分

 **整体二分的应用场**

1. 单个问题的答案具有单调性，当答案固定时，问题的要求可能得到满足或者无法满足
2. 如果要求得到满足，那么答案变差，要求依然能满足，此时去寻找更好的答案
3. 如果要求无法满足，那么答案变好，要求也无法满足，此时去寻找更差的答案
4. 问题数量较多，无法对每个问题都执行一次二分答案，但是所有查询允许**离线处理**

**整体二分的基本思想**

1. 让修改操作适应当前的答案范围，调整好数据状况，然后查看每个问题的要求是否达标
2. **多个问题共享这一轮的修改操作后的数据状况，避免重复劳动**
3. 达标的问题去往左组，不达标的问题去往右组，然后左右分别递归，直到解决所有的问题

**整体二分的时间复杂度**

​	假设，答案范围为`n`，问题数量为`q`，单次修改或查询为`log n`

​	那么，总的时间复杂度为O((n + q) * log n * log n)

​	O((n + q) * log n ***数据结构的单次修改或查询复杂度**)

```cpp
vector<int> maxSlidingWindow(vector<int>& nums, int k) {
    vector<int> res;
    deque<int> q{0};  // 存储下标，队首为当前窗口最大值的下标

        while (!q.empty() && nums[i] >= nums[q.back()]) {

            res.push_back(nums[q.front()]);
        }
    }
```

```c++
class Solution {
public:
    // 标准的二分搜索框架，搜索目标元素的索引，若不存在则返回 -1
    int search(vector<int>& nums, int target) {
        int left = 0;
        // 注意
        int right = nums.size() - 1;

        while(left <= right) {
            int mid = left + (right - left) / 2;
            if(nums[mid] == target) {
                return mid;   
            } else if (nums[mid] < target) {
                // 注意
                left = mid + 1;
            } else if (nums[mid] > target) {
                // 注意
                right = mid - 1;
            }
        }
        return -1;
    }
};
```

### 2.寻找左侧边界的二分搜索

```c++
//相当于lower_bound()函数   返回第一个大于等于（>=）val的位置
int left_bound(vector<int>& nums, int target) {
    int left = 0, right = nums.size() - 1;
    // 搜索区间为 [left, right]
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (nums[mid] < target) {
            // 搜索区间变为 [mid+1, right]              //不要r=m 或 l=m 不然死循环
            left = mid + 1;
        } else if (nums[mid] > target) { 
            // 搜索区间变为 [left, mid-1]
            right = mid - 1;
        } else if (nums[mid] == target) {
            // 收缩右侧边界
            right = mid - 1;
        }
    }
    // 判断 target 是否存在于 nums 中					
    // 如果越界，target 肯定不存在，返回 -1
    if (left < 0 || left >= nums.size()) {
        return -1;
    }
    // 判断一下 nums[left] 是不是 target
    return nums[left] == target ? left : -1;
}
```

### 3.寻找右侧边界的二分搜索

**如果 `target` 不存在，搜索右侧边界的二分搜索返回的索引是小于 `target` 的最大索引**。



```c++
int right_bound(vector<int>& nums, int target) {
    int left = 0, right = nums.size() - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (nums[mid] < target) {
            left = mid + 1;
        } else if (nums[mid] > target) {		 //不要r=m 或 l=m 不然死循环
            right = mid - 1;
        } else if (nums[mid] == target) {
            // 这里改成收缩左侧边界即可
            left = mid + 1;
        }
    }
    // 最后改成返回 left - 1
    if (left - 1 < 0 || left - 1 >= nums.size()) {       //最终结果    r+1=l
        return -1;
    }
	法1：return nums[left - 1] == target ? (left - 1) : -1;
当然，由于 while 的结束条件为right == left - 1所以你把上述代码中的 left - 1 都改成 right 也没有问题，这样可能更有利于看出来这是在「搜索右侧边界」：
    法2：return nums[right] == target ? right : -1;
}
```

```c++
// 在区间 [l, r] 中查找数值 val 出现的次数
int count(int val, int l, int r) {
    auto it_start = lower_bound(sorted_vals.begin(), sorted_vals.end(), make_pair(val, -1));
    if (it_start == sorted_vals.end() || it_start->first != val) return 0;
```
